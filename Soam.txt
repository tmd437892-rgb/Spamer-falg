# ---------------- TOKEN SYSTEM ---------------- #
def save_token(token):
    with open("token.json", "w") as file:
        json.dump({"TOKEN": token}, file)

def load_token():
    try:
        with open("token.json", "r") as file:
            data = json.load(file)
            return data.get("TOKEN")
    except:
        return None

# ---------------- ADMIN SYSTEM ---------------- #
def load_admin_users():
    """Load admin users from JSON file"""
    if os.path.exists(ADMIN_USERS_FILE):
        try:
            with open(ADMIN_USERS_FILE, "r") as f:
                return json.load(f)
        except:
            return []
    return []

def save_admin_users(users):
    """Save admin users to JSON file"""
    with open(ADMIN_USERS_FILE, "w") as f:
        json.dump(users, f, indent=2)

def add_admin_user(user_id):
    """Add a user to admin list"""
    users = load_admin_users()
    if user_id not in users:
        users.append(user_id)
        save_admin_users(users)
        print(f"âœ… User {user_id} added to admin users")
    return user_id in users

def is_admin(user_id):
    """Check if user is admin"""
    users = load_admin_users()
    return user_id in users or user_id == OWNER_ID

def remove_admin_user(user_id):
    """Remove a user from admin list"""
    users = load_admin_users()
    if user_id in users:
        users.remove(user_id)
        save_admin_users(users)
        print(f"âœ… User {user_id} removed from admin users")
    return user_id not in users
